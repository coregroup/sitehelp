Declaração de variável incompatível com seu tipo declarado



Um erro que acontece quando erramos o tipo de uma variável.


Exemplos:

ERRADO

 

//Escreva um algoritmo que leia 4 notas de um aluno e depois informe a média. 
programa
{
	funcao inicio()
	{

		inteiro nota, cont=0, i // A variável "nota" e "cont" deveriam ser reais, 
					//afinal um aluno pode tirar tanto de nota(nota) como de média(cont): 5.5, 9.7, etc.

		para (i=1; i<=4; i++) {
			escreva  ("Informe a ", i," nota do aluno :")
			leia (nota)
			cont = nota+cont	
		}

		escreva  ("A média do aluno é : ", cont/4)
	}
}

CERTO

 
//Descricao do programa: Escreva um algoritmo que leia 4 notas de um aluno e depois informe a média. 
programa
{
	funcao inicio()
	{
		inteiro i
		real nota, cont=0

		para (i=1; i<=4; i++) {
			escreva  ("Informe a ", i," nota do aluno :")
			leia (nota)
			cont = nota+cont	
		}

		escreva  ("A média do aluno é : ", cont/4)
	}
}
