Inicio Contatos Duvidas Sobre

Colocar espaço entre os dois caracteres que compoem os operadores ==, <=, >=, !=.



Exemplos:

ERRADO

 
//Descricao do programa: Informar todos os números pares no intervalo entre 100 e 200. 
programa
{
	funcao inicio()
	{		
		inteiro i 	
		para(i=100;i< =200;i++){  //Não pode haver espaço entre as expressões ==, <=, >=, !=.
		se(i%2==0){  
		escreva(i,"\n")
			}
		}
	  }
}

CERTO

 
//Descricao do programa: Informar todos os números pares no intervalo entre 100 e 200. 
programa
{
	funcao inicio()
	{		
		inteiro i 	
		para(i=100;i<=200;i++){  //Forma correta de usar as expressões, sem espaço entre elas: ==, <=, >=, !=.
		se(i%2==0){  
		escreva(i,"\n")
			}
		}
	  }
}
