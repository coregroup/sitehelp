Errar a escrita de um comando.



Exemplos:

ERRADO

 
//Descricao do programa: Leia o nome, altura, peso e idade do usuario. 
programa
{
	funcao inicio()
	{		
		cadeia nome 	
		real altura, peso
		inteiro idade
		escreve("Digite seu nome: ")//Escrita incorreta do comando "escreva".
		leia(nome)
		escreva("Digite sua idade: ")
		leia(idade)
		escreva("Digite sua altura: ")
		leia(altura)
		escreva("Digite seu peso: ")
		leya(peso)//Escrita incorreta do comando "leia".
	}
}
